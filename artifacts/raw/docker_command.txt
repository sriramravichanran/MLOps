# Global Jenkins for project
1. docker exec -u root -it jenkins-dind bash
2. apt-get update
3. apt-get install -y curl apt-transport-https ca-certificates gnupg
4. curl https://packages.cloud.google.com/apt/doc/apt-key.gpg | apt-key add -


If your getting error on the above 4th command
1. mkdir -p /etc/apt/keyrings
2. curl -fsSL https://packages.cloud.google.com/apt/doc/apt-key.gpg \
| gpg --dearmor -o /etc/apt/keyrings/google.gpg
3. echo "deb [signed-by=/etc/apt/keyrings/google.gpg] https://apt.kubernetes.io/ kubernetes-xenial main" \
> /etc/apt/sources.list.d/kubernetes.list
4. apt-get update

If your getting error on the above 4th command
1. curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
2. install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl
3. kubectl version --client
4. echo "deb https://packages.cloud.google.com/apt cloud-sdk main" | tee -a /etc/apt/sources.list.d/google-cloud-sdk.list
5. apt-get update && apt-get install -y google-cloud-sdk
6. apt-get update && apt-get install -y kubectl
7. kubectl version --client
8. gcloud --version

If gcloud is not installed proper run the below commands
1. cd /tmp
2. curl -O https://dl.google.com/dl/cloudsdk/channels/rapid/downloads/google-cloud-cli-linux-x86_64.tar.gz
3. tar -xf google-cloud-cli-linux-x86_64.tar.gz
4. ./google-cloud-sdk/install.sh --quiet
export PATH=$PATH:/tmp/google-cloud-sdk/bin
5. export PATH=$PATH:/tmp/google-cloud-sdk/bin
6. echo 'export PATH=$PATH:/tmp/google-cloud-sdk/bin' >> ~/.bashrc
source ~/.bashrc
7. gcloud --version
8. exit
